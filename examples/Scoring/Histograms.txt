s:Ge/Phantom/Parent="World"

b:Sc/DoseAtPhantom/OutputToConsole = "TRUE"
s:Sc/DoseAtPhantom/IfOutputFileAlreadyExists = "Overwrite"

i:Sc/DoseAtPhantom/HistogramBins = 10
